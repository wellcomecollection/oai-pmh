 80    def _determine_granularity(self, dt: datetime) -> str:
 81        if self.datestamp_granularity == "auto":
 82            has_time_component = any(
 83                (dt.hour, dt.minute, dt.second, dt.microsecond)
 84            )
 85            return "YYYY-MM-DDThh:mm:ssZ" if has_time_component else "YYYY-MM-DD"
 86        return self.datestamp_granularity

 88    def _format_datestamp(self, dt: Datestamp) -> str:
 89        """
 90        Formats a datetime object into an OAI-PMH datestamp string.
 91        """
 92        if isinstance(dt, str):
 93            return dt
 94        if dt.tzinfo is None:
 95            # If the datetime object is naive, assume it's in UTC.
 96            dt = dt.replace(tzinfo=timezone.utc)
 97        # If the datetime object is aware, convert it to UTC.
 98        dt = dt.astimezone(timezone.utc)
 99        granularity = self._determine_granularity(dt)
100        if granularity == "YYYY-MM-DD":
101            return dt.strftime("%Y-%m-%d")
102        return dt.strftime("%Y-%m-%dT%H:%M:%SZ")

104    def _request(self, verb: str, **kwargs) -> etree._Element:
105        """
106        Makes a request to the OAI-PMH repository and returns the parsed XML.
107
108        :param verb: The OAI-PMH verb.
109        :param kwargs: Additional request parameters.
110        :return: The parsed XML response.
111        """
112        params = {"verb": verb}
113        # Filter out None values so they aren't included in the query
114        for key, value in kwargs.items():
115            if value is not None:
116                params[key] = value
117
118        if self.use_post:
119            response = self._client.post(self.base_url, data=params)
120        else:
121            response = self._client.get(self.base_url, params=params)
122
123        response.raise_for_status()
124        xml = etree.fromstring(response.content)
125
126        error = xml.find("oai:error", namespaces=NS)
127        if error is not None:
128            code = error.get("code", "")
129            message = error.text or ""
130            exception_class = OAI_ERROR_MAP.get(code, OAIError)
131            raise exception_class(message)
132
133        return xml

161        """

163        """

165        verb = "ListSets"
