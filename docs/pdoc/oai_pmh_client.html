 90    def _determine_granularity(self, dt: datetime) -> str:
 91        if self.datestamp_granularity == "auto":
 92            has_time_component = any((dt.hour, dt.minute, dt.second, dt.microsecond))
 93            return "YYYY-MM-DDThh:mm:ssZ" if has_time_component else "YYYY-MM-DD"
 94        return self.datestamp_granularity

 96    def _determine_granularity_for_dates(
 97        self, dates: Sequence[Datestamp | None]
 98    ) -> str:
 99        if self.datestamp_granularity != "auto":
100            return self.datestamp_granularity
101
102        for dt in dates:
103            if isinstance(dt, datetime):
104                # Ensure we check granularity against UTC time
105                if dt.tzinfo is None:
106                    dt = dt.replace(tzinfo=timezone.utc)
107                dt = dt.astimezone(timezone.utc)
108
109                if any((dt.hour, dt.minute, dt.second, dt.microsecond)):
110                    return "YYYY-MM-DDThh:mm:ssZ"
111        return "YYYY-MM-DD"

113    def _format_datestamp(self, dt: Datestamp, granularity: str | None = None) -> str:
114        """
115        Formats a datetime object into an OAI-PMH datestamp string.
116        """
117        if isinstance(dt, str):
118            return dt
119        if dt.tzinfo is None:
120            # If the datetime object is naive, assume it's in UTC.
121            dt = dt.replace(tzinfo=timezone.utc)
122        # If the datetime object is aware, convert it to UTC.
123        dt = dt.astimezone(timezone.utc)
124
125        if granularity is None:
126            granularity = self._determine_granularity(dt)
127
128        if granularity == "YYYY-MM-DD":
129            return dt.strftime("%Y-%m-%d")
130        return dt.strftime("%Y-%m-%dT%H:%M:%SZ")

169    def _send_with_retries(self, request: httpx.Request) -> httpx.Response:
170        attempt = 0
171        while True:
172            attempt += 1
173            try:
174                return self._client.send(request)
175            except (httpx.ReadTimeout, httpx.ConnectTimeout, httpx.WriteTimeout):
176                if attempt >= self.max_request_retries:
177                    raise
178                delay = min(
179                    self.request_backoff_factor * (2 ** (attempt - 1)),
180                    self.request_max_backoff,
181                )
182                logger.warning(
183                    "Request timeout (attempt %s/%s). Retrying in %.2fs...",
184                    attempt,
185                    self.max_request_retries,
186                    delay,
187                )
188                if delay > 0:
189                    time.sleep(delay)
